CREATE TABLE areas (
  radius INT,
  area DECIMAL(14,4)
);

DELIMITER $$

CREATE PROCEDURE calc_areas()
BEGIN
  DECLARE v_radius INT DEFAULT 5;
  DECLARE v_area DECIMAL(14,4);
  DECLARE v_pi DECIMAL(10,8) DEFAULT 3.14159265;

  DELETE FROM areas;

  WHILE v_radius <= 9 DO
    SET v_area = v_pi * POW(v_radius, 2);
    INSERT INTO areas (radius, area) VALUES (v_radius, v_area);
    SET v_radius = v_radius + 1;
  END WHILE;
END$$

DELIMITER ;

CALL calc_areas();

SELECT * FROM areas;
