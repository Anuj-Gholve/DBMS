-- Create tables
CREATE TABLE Companies (
    comp_id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    cost DECIMAL(10,2),
    year INT
);

CREATE TABLE Orders (
    comp_id INT,
    domain VARCHAR(50),
    quantity INT,
    FOREIGN KEY (comp_id) REFERENCES Companies(comp_id)
);

-- 1. INNER JOIN: Names, costs, domains and quantities for companies
SELECT c.name, c.cost, o.domain, o.quantity
FROM Companies c
INNER JOIN Orders o ON c.comp_id = o.comp_id;

-- 2. LEFT OUTER JOIN
SELECT c.name, c.cost, o.domain, o.quantity
FROM Companies c
LEFT JOIN Orders o ON c.comp_id = o.comp_id;

-- 3. RIGHT OUTER JOIN
SELECT c.name, c.cost, o.domain, o.quantity
FROM Companies c
RIGHT JOIN Orders o ON c.comp_id = o.comp_id;

-- 4. UNION operator (Combine left and right join results)
SELECT c.name, c.cost, o.domain, o.quantity
FROM Companies c
LEFT JOIN Orders o ON c.comp_id = o.comp_id
UNION
SELECT c.name, c.cost, o.domain, o.quantity
FROM Companies c
RIGHT JOIN Orders o ON c.comp_id = o.comp_id;

-- 5. Create View View1 (Company name and quantities)
CREATE VIEW View1 AS
SELECT c.name AS Company_Name, o.quantity AS Quantity
FROM Companies c
JOIN Orders o ON c.comp_id = o.comp_id;

-- 6. Create View View2 with two columns (cost, year)
CREATE VIEW View2 AS
SELECT cost, year FROM Companies;

-- Insert, Update, Delete operations on View2
INSERT INTO View2 VALUES (25000, 2022);
UPDATE View2 SET cost = 26000 WHERE year = 2022;
DELETE FROM View2 WHERE year = 2022;

-- 7. Display content of View1 and View2
SELECT * FROM View1;
